// Graph state
const graphState = {
  nodes: new Map(),
  edges: [],
  layout: 'force',
  zoom: 1,
  pan: { x: 0, y: 0 },
  selectedNode: null,
  simulation: null,
  minWeight: 0
};

// Update graph layout
function updateGraphLayout(layout) {
  graphState.layout = layout;
  renderCausalGraphSVG();
  logToConsole('info', `Layout changed to: ${layout}`);
}

// Filter edges by weight
function filterCausalEdges(weight) {
  document.getElementById('weight-filter-value').textContent = parseFloat(weight).toFixed(1);
  graphState.minWeight = parseFloat(weight);
  renderCausalGraphSVG();
  updateGraphMetrics();
}

// Update node sizes based on centrality
function updateNodeSizes(metric) {
  graphState.nodeSizeMetric = metric;
  renderCausalGraphSVG();
  logToConsole('info', `Node sizes updated based on: ${metric}`);
}

// Switch between different views
function switchGraphView(view) {
  // Hide all views
  document.getElementById('graph-svg-container').style.display = 'none';
  document.getElementById('graph-list-view').style.display = 'none';

  }
}

// Render SVG graph visualization
async function renderCausalGraphSVG() {
  try {
    const edges = await getCausalEdges();
    if (edges.length === 0) return;

    // Build graph data structure
    graphState.nodes.clear();
    graphState.edges = edges.filter(e => {
      const metadata = typeof e.metadata === 'string' ? JSON.parse(e.metadata || '{}') : (e.metadata || {});
      const weight = metadata.weight || 0.5;
      return weight >= graphState.minWeight;
    });

    // Extract unique nodes
    graphState.edges.forEach(edge => {
      if (!graphState.nodes.has(edge.cause)) {
        graphState.nodes.set(edge.cause, { id: edge.cause, label: edge.cause });
      }
      if (!graphState.nodes.has(edge.effect)) {
        graphState.nodes.set(edge.effect, { id: edge.effect, label: edge.effect });
      }
    });

    const nodes = Array.from(graphState.nodes.values());

    // Calculate layout positions
    const positions = calculateLayout(nodes, graphState.edges, graphState.layout);

    // Render SVG
    const svg = document.getElementById('causal-graph-svg');
    const container = document.getElementById('graph-container');
    container.innerHTML = '';

    const svgRect = svg.getBoundingClientRect();
    const width = svgRect.width;
    const height = svgRect.height;

    // Draw edges
    graphState.edges.forEach(edge => {
      const source = positions.get(edge.cause);
      const target = positions.get(edge.effect);
      if (!source || !target) return;

      const metadata = typeof edge.metadata === 'string' ? JSON.parse(edge.metadata || '{}') : (edge.metadata || {});
      const weight = metadata.weight || 0.5;

      const line = document.createElementNS('http://www.w3.org/2000/svg', 'line');
      line.setAttribute('x1', source.x);
      line.setAttribute('y1', source.y);
      line.setAttribute('x2', target.x);
      line.setAttribute('y2', target.y);
      line.setAttribute('stroke', '#00ff88');
      line.setAttribute('stroke-width', 1 + weight * 3);
      line.setAttribute('stroke-opacity', 0.3 + weight * 0.7);
      line.setAttribute('marker-end', 'url(#arrowhead)');
      container.appendChild(line);
    });

    // Draw nodes
    nodes.forEach(node => {
      const pos = positions.get(node.id);
      if (!pos) return;

      const g = document.createElementNS('http://www.w3.org/2000/svg', 'g');
      g.setAttribute('cursor', 'pointer');
      g.onclick = () => selectNode(node.id);

      const circle = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
      circle.setAttribute('cx', pos.x);
      circle.setAttribute('cy', pos.y);
      circle.setAttribute('r', getNodeSize(node.id));
      circle.setAttribute('fill', graphState.selectedNode === node.id ? '#00ff88' : '#2a4a3a');
      circle.setAttribute('stroke', '#00ff88');
      circle.setAttribute('stroke-width', '2');

      const text = document.createElementNS('http://www.w3.org/2000/svg', 'text');
      text.setAttribute('x', pos.x);
      text.setAttribute('y', pos.y + 25);
      text.setAttribute('text-anchor', 'middle');
      text.setAttribute('fill', '#e0e0e0');
      text.setAttribute('font-size', '10');
      text.textContent = node.label.substring(0, 20) + (node.label.length > 20 ? '...' : '');

      g.appendChild(circle);
      g.appendChild(text);
      container.appendChild(g);
    });

    updateGraphMetrics();

  } catch (error) {
    logToConsole('error', `Failed to render graph: ${error.message}`);
  }
}

// Calculate layout positions
function calculateLayout(nodes, edges, layout) {
  const positions = new Map();
  const width = 700;
  const height = 500;
  const centerX = width / 2;
  const centerY = height / 2;

  switch(layout) {
    case 'force':
      // Simple force-directed layout
      nodes.forEach((node, i) => {
        const angle = (i / nodes.length) * 2 * Math.PI;
        const radius = Math.min(width, height) / 3;
        positions.set(node.id, {
          x: centerX + radius * Math.cos(angle),
          y: centerY + radius * Math.sin(angle)
        });
      });
      break;

    case 'hierarchical':
      // Hierarchical layout
      const levels = new Map();
      nodes.forEach(node => levels.set(node.id, 0));

      // Calculate levels based on edges
      edges.forEach(edge => {
        const sourceLevel = levels.get(edge.cause) || 0;
        const targetLevel = levels.get(edge.effect) || 0;
        if (targetLevel <= sourceLevel) {
          levels.set(edge.effect, sourceLevel + 1);
        }
      });

      const maxLevel = Math.max(...Array.from(levels.values()));
      const nodesPerLevel = new Map();

      levels.forEach((level, nodeId) => {
        if (!nodesPerLevel.has(level)) nodesPerLevel.set(level, []);
        nodesPerLevel.get(level).push(nodeId);
      });

      nodesPerLevel.forEach((nodeIds, level) => {
        nodeIds.forEach((nodeId, i) => {
          positions.set(nodeId, {
            x: (i + 1) * width / (nodeIds.length + 1),
            y: 50 + (level * (height - 100) / (maxLevel || 1))
          });
        });
      });
      break;

    case 'circular':
      // Circular layout
      nodes.forEach((node, i) => {
        const angle = (i / nodes.length) * 2 * Math.PI;
        const radius = Math.min(width, height) / 2.5;
        positions.set(node.id, {
          x: centerX + radius * Math.cos(angle),
          y: centerY + radius * Math.sin(angle)
        });
      });
      break;

    case 'radial':
      // Radial layout
      const roots = nodes.filter(node =>
        !edges.some(e => e.effect === node.id)
      );

      if (roots.length > 0) {
        const root = roots[0];
        positions.set(root.id, { x: centerX, y: centerY });

        const levels = new Map([[root.id, 0]]);
        const queue = [root.id];

        while (queue.length > 0) {
          const current = queue.shift();
          const currentLevel = levels.get(current);
          const children = edges.filter(e => e.cause === current).map(e => e.effect);

          children.forEach((child, i) => {
            if (!levels.has(child)) {
              levels.set(child, currentLevel + 1);
              queue.push(child);

              const angle = (i / children.length) * 2 * Math.PI;
              const radius = (currentLevel + 1) * 80;
              positions.set(child, {
                x: centerX + radius * Math.cos(angle),
                y: centerY + radius * Math.sin(angle)
              });
            }
          });
        }
      }
      break;
  }

  return positions;
}

// Get node size based on metric
function getNodeSize(nodeId) {
  const metric = graphState.nodeSizeMetric || 'uniform';
  const baseSize = 8;

  switch(metric) {
    case 'degree':
      const degree = getDegree(nodeId);
      return baseSize + degree * 2;

    case 'betweenness':
      const betweenness = calculateBetweenness(nodeId);
      return baseSize + betweenness * 10;

    case 'pagerank':
      const pagerank = calculatePageRank(nodeId);
      return baseSize + pagerank * 15;

    default:
      return baseSize;
  }
}

// Calculate node degree
function getDegree(nodeId) {
  return graphState.edges.filter(e =>
    e.cause === nodeId || e.effect === nodeId
  ).length;
}

// Calculate betweenness centrality (simplified)
function calculateBetweenness(nodeId) {
  let betweenness = 0;
  const nodes = Array.from(graphState.nodes.keys());

  nodes.forEach(source => {
    nodes.forEach(target => {
      if (source !== target && source !== nodeId && target !== nodeId) {
        const paths = findAllPaths(source, target);
        const pathsThroughNode = paths.filter(path => path.includes(nodeId));
        if (paths.length > 0) {
          betweenness += pathsThroughNode.length / paths.length;
        }
      }
    });
  });

  return betweenness / (nodes.length * nodes.length);
}

// Calculate PageRank (simplified)
function calculatePageRank(nodeId) {
  const dampingFactor = 0.85;
  const nodes = Array.from(graphState.nodes.keys());
  const ranks = new Map(nodes.map(n => [n, 1 / nodes.length]));

  // Simple iterative calculation
  for (let i = 0; i < 10; i++) {
    const newRanks = new Map();

    nodes.forEach(node => {
      let rank = (1 - dampingFactor) / nodes.length;

      const incomingEdges = graphState.edges.filter(e => e.effect === node);
      incomingEdges.forEach(edge => {
        const outDegree = graphState.edges.filter(e => e.cause === edge.cause).length;
        rank += dampingFactor * (ranks.get(edge.cause) / outDegree);
      });

      newRanks.set(node, rank);
    });

    newRanks.forEach((rank, node) => ranks.set(node, rank));
  }

  return ranks.get(nodeId) || 0;
}

// Find all paths between two nodes
function findAllPaths(source, target, visited = new Set()) {
  if (source === target) return [[source]];

  visited.add(source);
  const paths = [];

  const outEdges = graphState.edges.filter(e => e.cause === source);
  outEdges.forEach(edge => {
    if (!visited.has(edge.effect)) {
      const subPaths = findAllPaths(edge.effect, target, new Set(visited));
      subPaths.forEach(subPath => {
        paths.push([source, ...subPath]);
      });
    }
  });

  return paths;
}

Effects (${outEdges.length})

  `;

// Update graph metrics
function updateGraphMetrics() {
  const nodes = Array.from(graphState.nodes.keys());
  const edges = graphState.edges;

  const numNodes = nodes.length;
  const numEdges = edges.length;

  // Calculate density
  const maxEdges = numNodes * (numNodes - 1);
  const density = maxEdges > 0 ? (numEdges / maxEdges) : 0;

  // Calculate average path length
  let totalPathLength = 0;
  let pathCount = 0;

  nodes.forEach(source => {
    nodes.forEach(target => {
      if (source !== target) {
        const paths = findAllPaths(source, target);
        if (paths.length > 0) {
          const shortestPath = paths.reduce((min, path) =>
            path.length < min.length ? path : min
          );
          totalPathLength += shortestPath.length - 1;
          pathCount++;
        }
      }
    });
  });

  const avgPathLength = pathCount > 0 ? (totalPathLength / pathCount) : 0;

  // Calculate clustering coefficient
  const clustering = calculateClusteringCoefficient();

  // Find strongly connected components
  const components = findStronglyConnectedComponents();

  document.getElementById('metric-nodes').textContent = numNodes;
  document.getElementById('metric-edges').textContent = numEdges;
  document.getElementById('metric-density').textContent = density.toFixed(2);
  document.getElementById('metric-avg-path').textContent = avgPathLength.toFixed(2);
  document.getElementById('metric-clustering').textContent = clustering.toFixed(2);
  document.getElementById('metric-components').textContent = components.length;
}

// Calculate clustering coefficient
function calculateClusteringCoefficient() {
  const nodes = Array.from(graphState.nodes.keys());
  let totalCoefficient = 0;

  nodes.forEach(node => {
    const neighbors = new Set();
    graphState.edges.forEach(edge => {
      if (edge.cause === node) neighbors.add(edge.effect);
      if (edge.effect === node) neighbors.add(edge.cause);
    });

    const neighborArray = Array.from(neighbors);
    if (neighborArray.length < 2) return;

    let connections = 0;
    neighborArray.forEach((n1, i) => {
      neighborArray.slice(i + 1).forEach(n2 => {
        if (graphState.edges.some(e =>
          (e.cause === n1 && e.effect === n2) || (e.cause === n2 && e.effect === n1)
        )) {
          connections++;
        }
      });
    });

    const maxConnections = neighborArray.length * (neighborArray.length - 1) / 2;
    totalCoefficient += maxConnections > 0 ? (connections / maxConnections) : 0;
  });

  return nodes.length > 0 ? (totalCoefficient / nodes.length) : 0;
}

// Find strongly connected components (Tarjan's algorithm simplified)
function findStronglyConnectedComponents() {
  const nodes = Array.from(graphState.nodes.keys());
  const visited = new Set();
  const components = [];

  nodes.forEach(node => {
    if (!visited.has(node)) {
      const component = new Set();
      const stack = [node];

      while (stack.length > 0) {
        const current = stack.pop();
        if (visited.has(current)) continue;

        visited.add(current);
        component.add(current);

        graphState.edges.forEach(edge => {
          if (edge.cause === current && !visited.has(edge.effect)) {
            stack.push(edge.effect);
          }
        });
      }

      components.push(Array.from(component));
    }
  });

  return components;
}

// Render adjacency matrix
async function renderAdjacencyMatrix() {
  const edges = await getCausalEdges();
  const nodes = new Set();

  edges.forEach(edge => {
    nodes.add(edge.cause);
    nodes.add(edge.effect);
  });

  const nodeArray = Array.from(nodes);
  const container = document.getElementById('adjacency-matrix');

  if (nodeArray.length === 0) {
    container.innerHTML = `
      

        
📊

        
No nodes to display

      

    `;
    return;
  }

  let html = '';

  // Header row
  html += '';
  nodeArray.forEach(node => {
    html += ``;
  });
  html += '';

  // Data rows
  nodeArray.forEach(source => {
    html += ``;

    nodeArray.forEach(target => {
      const edge = edges.find(e => e.cause === source && e.effect === target);
      const metadata = edge && typeof edge.metadata === 'string' ? JSON.parse(edge.metadata || '{}') : (edge?.metadata || {});
      const weight = metadata.weight || 0;

      const bgColor = weight > 0 ? `rgba(0, 255, 136, ${weight})` : 'transparent';
      html += ``;
    });

    html += '';
  });

  html += '
${node.substring(0, 15)}
${source.substring(0, 15)}${weight > 0 ? weight.toFixed(2) : '-'}
';
  container.innerHTML = html;
}

// Get causal edges from database
async function getCausalEdges() {
  try {
    const tableExists = sqlAll("SELECT name FROM sqlite_master WHERE type='table' AND name='causal_edges'");
    if (tableExists.length === 0) return [];

    return sqlAll('SELECT * FROM causal_edges ORDER BY created_at DESC LIMIT 100');
  } catch (error) {
    console.error('Error getting causal edges:', error);
    return [];
  }
}

// Analysis functions
function findCentralNodes() {
  const nodes = Array.from(graphState.nodes.keys());
  const centrality = nodes.map(nodeId => ({
    node: nodeId,
    pagerank: calculatePageRank(nodeId),
    betweenness: calculateBetweenness(nodeId),
    degree: getDegree(nodeId)
  }));

  centrality.sort((a, b) => b.pagerank - a.pagerank);

  const top5 = centrality.slice(0, 5);
  const message = top5.map((c, i) =>
    `${i + 1}. ${c.node.substring(0, 30)} (PageRank: ${c.pagerank.toFixed(3)})`
  ).join('\n');

  alert('🎯 Most Central Nodes:\n\n' + message);
  logToConsole('info', 'Central nodes analysis completed');
}

}

function findConfounders() {
  // Find nodes that are common causes of multiple other nodes
  const nodes = Array.from(graphState.nodes.keys());
  const confounders = [];

  nodes.forEach(node => {
    const effects = graphState.edges.filter(e => e.cause === node).map(e => e.effect);
    if (effects.length >= 2) {
      confounders.push({ node, effects: effects.length });
    }
  });

  confounders.sort((a, b) => b.effects - a.effects);

  if (confounders.length === 0) {
    alert('🔀 No potential confounders found');
  } else {
    const message = confounders.slice(0, 10).map((c, i) =>
      `${i + 1}. ${c.node.substring(0, 30)} → ${c.effects} effects`
    ).join('\n');
    alert('🔀 Potential Confounders:\n\n' + message);
  }

  logToConsole('info', `Found ${confounders.length} potential confounders`);
}
